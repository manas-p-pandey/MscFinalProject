PostgreSQL
---------------------
docker pull: docker pull postgres

docker run: docker run --name postgres \
  -e POSTGRES_USER=postgres \
  -e POSTGRES_PASSWORD=Admin123 \
  -e POSTGRES_DB=mscds \
  -p 5432:5432 \
  -d postgres

password: Admin123
url: http://localhost:5432
